def linearRegressionFunc(df):
    X = np.array(df['new_people_vaccinated_smoothed_per_hundred']).reshape(-1, 1)
    y = np.array(df['new_deaths_per_million']).reshape(-1, 1)
    regr = LinearRegression()
    regr.fit(X, y)
    y_pred = regr.predict(X)
    return (X, y, y_pred, regr.coef_, regr.intercept_)
